\begin{equation}
\label{eq:NL}
    \begin{gathered}
       x_{i+1} = x_i + V\cos{\theta}\Delta t + \nu_x  \\
       y_{i+1} = y_i + V\sin{\theta}\Delta t  + \nu_y\\ 
       z_{i+1} = z_i + \bar{v_z}\Delta t + v_z\Delta t + \nu_z\\ 
       \theta_{i+1} = \theta_i + \omega\Delta t + \nu_{\theta} 
    \end{gathered}
\end{equation}
